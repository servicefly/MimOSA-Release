# ===========================================================================
# MimOSA - Phase 1 Python dependencies
# Target: Python 3.10+  (developed/tested on 3.11)
#
# Install with:   pip install -r requirements.txt
#
# Notes:
#   * pyaudio requires the PortAudio system library:
#         sudo apt install portaudio19-dev
#   * openai-whisper pulls in PyTorch (large download).
#   * Version pins use compatible-release / lower bounds to stay reproducible
#     while allowing patch updates.
# ===========================================================================

# --- LLM / API ---
requests>=2.31,<3.0            # HTTP client for Abacus.AI RouteLLM calls
python-dotenv>=1.0,<2.0        # Load configuration from .env files

# --- Voice: wake word ---
pvporcupine>=3.0,<4.0          # Porcupine wake-word detection

# --- Voice: speech-to-text (local) ---
openai-whisper>=20231117       # Local STT (no cloud dependency)

# --- Voice: text-to-speech (local) ---
piper-tts>=1.2,<2.0            # Local, natural-sounding TTS

# --- Audio I/O ---
pyaudio>=0.2.13,<0.3           # Microphone capture / playback (needs PortAudio)

# --- System integration ---
psutil>=5.9,<8.0               # CPU / RAM / process monitoring; app process control (M2.2)
send2trash>=1.8,<2.2           # Cross-platform "move to Trash" for safe deletes (M2.1)

# --- UI: GTK4 desktop avatar (M3.1) ---
# OPTIONAL. The avatar is graceful-degradation: MimOSA runs headless (voice/CLI)
# when these are absent. PyGObject also needs system packages (Kubuntu/Debian):
#     sudo apt install python3-gi gir1.2-gtk-4.0 libgirepository1.0-dev \
#                      libcairo2-dev gobject-introspection
# On older typelibs (GLib < 2.80, e.g. Debian 12) pin PyGObject to <3.52.
PyGObject>=3.42,<3.52          # GTK4 bindings for the avatar window (optional)
pycairo>=1.20                  # Cairo drawing for avatar animations (optional)

# --- Memory & Context: semantic recall (M5.3) ---
# OPTIONAL. Semantic memory degrades gracefully: when these are absent MimOSA
# falls back to a pure-Python, dependency-free embedder + in-process vector
# store (deterministic, fully offline), so conversation recall still works —
# just lexically rather than with learned embeddings. Install these for true
# on-device semantic recall:
#     pip install "chromadb>=0.4,<0.6" "sentence-transformers>=2.2,<4.0"
# sentence-transformers pulls in PyTorch (large download). Everything runs
# locally; no data ever leaves the device.
# chromadb>=0.4,<0.6           # Local vector store (optional, M5.3)
# sentence-transformers>=2.2   # On-device embeddings (optional, M5.3)

# --- Memory & Context: encrypted private store (M5.4, optional/future) ---
# The Privacy Guard keeps sensitive turns on-device today via the is_private
# flag + local-only routing. An encrypted SQLCipher private store is optional:
# sqlcipher3-binary>=0.5        # Encrypted private.db (optional, M5.4)

# --- Testing ---
pytest>=8.0,<9.0               # Test runner for the automated test suite
